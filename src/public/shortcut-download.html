Power users: copy/paste payload

Paste this into your browser console on any page where Suka is loaded. It creates a daily cleaning plan and a weekly garden plan and stores them as user schedules (not system). 

window.__suka?.eventBus?.emit?.("automation.schedule.request", {
  title: "Daily Cleaning – Landing Zones",
  templateId: "cleaning.session.generate",
  rule: { at: "09:00" },
  meta: { domain: "cleaning", source: "shortcut-download" },
  ctx: { focus: "entry,mudroom,5-bin" }
});

window.__suka?.eventBus?.emit?.("automation.schedule.request", {
  title: "Weekly Garden Care – Beds A/B",
  templateId: "garden.session.generate",
  rule: { at: "08:00", days: [6] }, // Saturday
  meta: { domain: "garden", source: "shortcut-download" },
  ctx: { beds: ["A","B"], harvestTo: "storehouse" }
});
